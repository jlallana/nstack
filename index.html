using System;
using NStack;
using System.Diagnostics;

//Service description
interface ILogger
{
	void Info(string message);
}

//Service helper (Optional for readability)
static class logger
{
	public static ILogger Current
	{
        [DebuggerHidden]
		get
		{
			return Context.Resolve<ILogger> ();
		}
	}

    [DebuggerHidden]
	public static void Info(string message)
	{
		Current.Info (message);
	}
}

//Console service implementation
class ConsoleLogger : ILogger
{
	public void Info(string message)
	{
		Console.WriteLine ("[INFO - {0}] {1}", DateTime.Now, message);
	}
}

//Void service implementation
class VoidLogger : ILogger
{
	public void Info(string message)
	{
	}
}

//Consumer class
class Server
{
	public void Start()
	{
		//Using located service
		logger.Info ("Server started!!!");
	}
}

class Program
{
	public static void Main (string[] args)
	{
		var server = new Server ();

		//Starts calling to a console logger
		Context.Create (() => {

			Context.Register<ILogger>(new ConsoleLogger());

			//Must puts message in console
			server.Start();

		});

		//Start calling to a void logger
		Context.Create (() => {

			Context.Register<ILogger>(new VoidLogger());

			//Must do nothing
			server.Start();

		});
	}
}
